set term png
set key top left
set fit quiet
set fit logfile '/dev/null'
f(x) = m*x + b

set output "output/contraction/courbes/contraction_cycle.png"
set title "Efficacité de l'opération de contraction"
set xlabel 'Taille du graphe cycle'
set ylabel 'Temps CPU (s)'
fit f(x) "output/contraction/data/cycle.dat" using 1:2 via m,b
plot "output/contraction/data/cycle.dat" using 1:2 with points title "Mesures", \
f(x) title "Approximation"
